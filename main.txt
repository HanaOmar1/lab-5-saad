#include <iostream>
using namespace std;

// Function to sum two integers
int sumde(int a, int b) {
    return a + b;
}

// Function to sum three integers
int sumde(int a, int b, int c) {
    return a + b + c;
}

int main() {
    // Example usage
    int result1 = sumde(2, 3); // Calls the sum function with 2 arguments
    int result2 = sumde(1, 4, 5); // Calls the sum function with 3 arguments

    cout << "Sum of 2 and 3: " << result1 << endl;
    cout << "Sum of 1, 4, and 5: " << result2 << endl;

    return 0;
}
